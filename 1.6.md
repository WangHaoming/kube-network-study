```
modprobe ipip
lsmod  | grep ipip

ip netns add ns1
ip netns add ns2
ip link add v1 type veth peer name vp1
ip link add v2 type veth peer name vp2
ip addr add 10.10.10.1/24 dev v1
ip addr add 10.10.20.1/24 dev v2
ip link set v1 up
ip link set v2 up


ip link set vp1 netns ns1
ip netns exec ns1 ip addr add 10.10.10.2/24 dev vp1
ip netns exec ns1 ip link set vp1 up


ip link set vp2 netns ns2
ip netns exec ns2 ip addr add 10.10.20.2/24 dev vp2
ip netns exec ns2 ip link set vp2 up
```

add route in ns1
```
ip netns exec ns1 route add -net 10.10.20.0 netmask 255.255.255.0 gw 10.10.10.1
```
